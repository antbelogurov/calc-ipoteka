## Модели данных (план)

```prisma
model User {
  id        String   @id @default(cuid())
  email     String   @unique
  password  String
  loans     Loan[]
  createdAt DateTime @default(now())
}

model Loan {
  id              String    @id @default(cuid())
  userId          String
  user            User      @relation(fields: [userId], references: [id])
  name            String    // "Ипотека Сбербанк" / "Кредит ВТБ"
  type            LoanType  // MORTGAGE | CREDIT
  amount          Float     // Сумма займа
  rate            Float     // Процентная ставка (годовых)
  termMonths      Int       // Срок в месяцах
  startDate       DateTime  // Дата начала
  monthlyPayment  Float     // Ежемесячный платёж
  payments        Payment[]
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
}

model Payment {
  id        String      @id @default(cuid())
  loanId    String
  loan      Loan        @relation(fields: [loanId], references: [id])
  amount    Float       // Сумма платежа
  date      DateTime    // Дата платежа
  type      PaymentType // REGULAR | OVERPAYMENT
  note      String?     // Комментарий
  createdAt DateTime    @default(now())
}

enum LoanType {
  MORTGAGE
  CREDIT
}

enum PaymentType {
  REGULAR
  OVERPAYMENT
}
```
